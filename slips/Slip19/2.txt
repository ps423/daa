Q.2) Write a program to show board configuration of 4 queens’ problem.

// 4queens.c 
#include <stdio.h> 
#include <math.h> 
int board[5]; // 1-indexed for rows 1..4 
int place(int row, int col) { 
    for (int i = 1; i < row; i++) { 
        if (board[i] == col || abs(board[i] - col) == abs(i - row)) 
            return 0; 
    } 
    return 1; 
} 
void printBoard() { 
    for (int i = 1; i <= 4; i++) { 
        for (int j = 1; j <= 4; j++) { 
            if (board[i] == j) 
                printf("Q "); 
            else 
                printf(". "); 
        } 
        printf("\n"); 
    } 
    printf("\n"); 
} 
void nQueens(int row) { 
    for (int col = 1; col <= 4; col++) { 
        if (place(row, col)) { 
            board[row] = col; 
            if (row == 4) 
                printBoard();  
            else 
                nQueens(row + 1); 
        } 
    } 
} 
int main() { 
    nQueens(1); 
    return 0; 
}
