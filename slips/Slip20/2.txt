Q.2) Write a program to solve N Queens Problem using Backtracking.

/ nqueens.c 
#include <stdio.h> 
#include <math.h> 
int board[20], n; 
int place(int row, int col) { 
    for (int i = 1; i < row; i++) { 
        if (board[i] == col || abs(board[i] - col) == abs(i - row)) 
            return 0; 
    } 
    return 1; 
} 
void printBoard() { 
    for (int i = 1; i <= n; i++) { 
        for (int j = 1; j <= n; j++) { 
            if (board[i] == j) 
                printf("Q "); 
            else 
                printf(". "); 
        } 
        printf("\n"); 
    } 
    printf("\n"); 
} 
void nQueens(int row) { 
    for (int col = 1; col <= n; col++) { 
        if (place(row, col)) { 
            board[row] = col; 
            if (row == n) 
                printBoard(); 
            else 
                nQueens(row + 1); 
        } 
    } 
} 
int main() { 
    printf("Enter number of queens: "); 
    scanf("%d", &n); 
    nQueens(1); 
    return 0;
}
